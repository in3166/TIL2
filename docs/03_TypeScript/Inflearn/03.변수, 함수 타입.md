// 문자열
const str: string = 'hello';

// 숫자
const num: number = 10;

// 배열
const arr: Array<string> = [];
arr.push('hi');

// 배열 - 리터럴 적용
const items: string[] = [];
// items.push(10);

// 튜플
const address: [string, number] = ['판교', 40];

// 객체
const obj: object = {};
// obj.a = 10;

// 타입 객체

person.age = 101;

  Capt,

  Hulk,

const myHero = Avengers.Capt;

a = 20;

  // return 'a';

let str = '이건 문자열 타입이 됩니다.';

# 함수 타입

```ts
// 자바스크립트 함수 선언
function sum(a, b) {
  return a + b;
}

// 타입스크립트 함수 선언 - 함수의 매개 변수
function add(a: number, b: number) {
  return a + b;
}

// 타입스크립트 함수 선언 - 함수의 반환 타입
function add(a: number, b: number): number {
  return a + b;
}

// function add(a: number, b: number): string {
//   return a + b;
// }

// 함수 인자
function log(a: string) {
  console.log(a);
}
log('a', 10);

// 함수 옵셔널 파라미터(optional parameter)
function printText(text: string, type?: string) {
  console.log(text);
}
printText('hi');
```
